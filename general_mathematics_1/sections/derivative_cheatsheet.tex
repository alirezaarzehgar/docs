\section{Derivative Cheat Sheet}
\begin{multicols}{2}
\begin{align*}
&(y = c)\to y' = 0 \\
&(ax^n)' = a(x^n)' = anx^{n-1} \\
&(u \pm v)' = u' \pm v' \\
&(uv)' = u'v + uv' \\
&(au^n)' = a(u^n)' = anu^{n-1}u' \\
&(\frac{u}{v})' = \frac{u'v - uv'}{v^2} \to (\frac{1}{x})' = \frac{-1}{x^2} \\
&(\sqrt{u})' = \frac{u'}{2\sqrt{u}} \to (\sqrt{x})' = \frac{1}{2\sqrt{x}} \\
\end{align*}

\end{multicols}
